\begin{exercises}

\ex{6.6.1}
Prove Proposition \thref{6.6.1}, \thref{6.6.2},
\thref{6.6.3}, and \thref{6.6.5}.

\ex{6.6.2}
Perform each of the indicated operations and write
the answer in the form $x + iy$.
\begin{exenum}
\x
$(3+i4)+(7-i3)$
\x
$(-2+i\sqrt2)+(-2-i\sqrt2)$
\x
$(-2+i\sqrt2)(-2-i\sqrt2)$
\x
$\frac{3-i7}{2+i}$
\x
$(a+ib)(2a-i2b)$
\x
$2(4-i3)+7(-2+i5)$
\x
$\frac{-3+i4}{4-i3}$
\x
$\frac{7+i}{-2-i5}$
\x
$\frac{a+ib}{3a-i3b}$
\x
$\frac{-2-i}{2-i}$
\x
$\frac{25}{3-i4}$
\x
$(2+i7)(2-i5)$.
\end{exenum}

\ex{6.6.3}
Let $z_1=2+i3$, $z_2=-1-i$, and $z_3=i$.
Plot each of the following complex numbers
in the complex plane.
\begin{exenum}
\x
$z_1$
\x
$z_2$
\x
$z_1+z_2$
\x
$z_1z_3$
\x
$z_1-2z_2$
\x
$\frac{z_1}{z_2}$.
\end{exenum}

\ex{6.6.4}
Find the complex conjugate of each of the following
complex numbers.
\begin{exenum}
\x
\xlab{6.6.4a}
$2-i3$
\x
\xlab{6.6.4b}
$5+i4$
\x
\xlab{6.6.4c}
$(2-i3)+(5+i4)$
\x
\xlab{6.6.4d}
$(2-i3)(5+i4)$
\x
\xlab{6.6.4e}
$4(2-i3)$
\x
$-7$
\x
$2i$.
\end{exenum}

\ex{6.6.5}
\begin{exenum}
\x
In Problem \exref{6.6.4},
compute the sum of conjugates formed in
\exref{6.6.4a} and \exref{6.6.4b},
and compare with the conjugate of the
sum in \exref{6.6.4c}.
\x
In Problem \exref{6.6.4},
compute the product of the conjugates found in
\exref{6.6.4a} and \exref{6.6.4b},
and compare with the conjugate of the product
found in \exref{6.6.4d}.
\x
In Problem \exref{6.6.4},
multiply the conjugate found in \exref{6.6.4a}
by $4$, and compare with the answer
found in \exref{6.6.4e}.
\end{exenum}

\ex{6.6.6}
For any complex numbers $z_1$ and $z_2$,
prove that
\begin{exenum}
\x
$\conj{z_1+z_2} = \conj{z_1} + \conj{z_2}$
\x
$\conj{z_1z_2} = \conj{z_1}\;\conj{z_2}$
\x
$\conj{kz_1} = k\conj{z_1}$, $k$ real.
\end{exenum}

\ex{6.6.7}
If $a$ is real and positive and $z$ complex,
prove that $|az| = a|z|$.

\ex{6.6.8}
\begin{exenum}
\x
Prove that the sum and product of two complex
numbers which are conjugates of each other
are real.
\x
Prove that the difference of two complex
numbers which are conjugates of each other
is pure imaginary.
\end{exenum}

\ex{6.6.9}
Graph all points $z$ satisfying
\begin{exenum}
\x
$|z|=2$
\x
$|z|<3$
\x
$|z|>1$
\x
$|z|\leq2$
\x
$2<|z|<4$
\x
$|z-2|=2$
\x
$|z-z_0|=3$, for a fixed $z_0$
\x
$1\leq|z-3|\leq2$.
\end{exenum}

\ex{6.6.10}
Given two complex numbers $z_1$ and $z_2$,
plot them and give a geometric interpretation
of $|z_1-z_2|$.

\ex{6.6.11}
\begin{exenum}
\x
Show that $-1$ has two square roots in $\C$.
\x
Show that every nonzero real number has two
square roots.
\end{exenum}

\end{exercises}
